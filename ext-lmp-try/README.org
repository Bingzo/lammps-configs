#+STARTUP: inlineimages

#+BEGIN_SRC gnuplot :results output :file vel.png
set macros
L=64.0
m = 1.0
vmax=3.09236030555307
b=0.5*L
g(x) = -(a * abs(b-x))**(m+1) + vmax
fq=0.1
limit="[fq*L:(1.0-fq)*L]"
file="<./spat.awk  vy.av.*"
fit @limit g(x)  file  via m, vmax, a, b

plot file w lp lw 3, g(x) t sprintf("m=%f", m)
#+END_SRC

#+results:
[[file:vel.png]]
